Web app plugins extend the Mattermost user interface by registering React components and hooks through the Plugin Registry. They run in the user's browser and can add UI elements throughout the application.

## Setup

### Project Structure

```
my-plugin/
├── webapp/
│   ├── src/
│   │   ├── index.tsx          # Main plugin entry point
│   │   ├── components/        # React components
│   │   ├── actions.ts         # Redux actions
│   │   └── reducers.ts        # Redux reducers
│   ├── package.json
│   ├── tsconfig.json
│   └── webpack.config.js
└── plugin.json                # Includes webapp.bundle_path
```

### Plugin Entry Point

Create `webapp/src/index.tsx`:

```typescript
import {Store} from 'redux';

// Define your plugin class
export default class Plugin {
    // Initialize is called when the plugin is loaded
    public initialize(registry: any, store: Store) {
        // Register your components and hooks here
        registry.registerPostTypeComponent('custom_mytype', MyCustomPost);
    }

    // Uninitialize is called when plugin is unloaded
    public uninitialize() {
        // Clean up if needed
    }
}

// Register the plugin
(window as any).registerPlugin('com.example.myplugin', new Plugin());
```

```javascript
const path = require('path');

module.exports = {

        },
    },
    module: {
        rules: [
            {
                test: /\.tsx?$/,
                use: 'ts-loader',
                exclude: /node_modules/,
            },
            {
                test: /\.scss$/,
                use: ['style-loader', 'css-loader', 'sass-loader'],
            },
        ],
    },
    externals: {
        react: 'React',
        redux: 'Redux',
        'react-redux': 'ReactRedux',
        'react-router-dom': 'ReactRouterDom',
    },
};
```

<Note>
  Mattermost provides React, Redux, and other common libraries. Mark them as `externals` to avoid bundling them.
</Note>

## Plugin Registry

The Plugin Registry provides methods to register UI components and hooks.

### Component Registration

<Tabs>
  <Tab title="Channel Header">
    Add buttons or icons to the channel header:

    ```typescript
    // Register a channel header button
    registry.registerChannelHeaderButtonAction(
        <i className="icon fa fa-gear"/>,
        (channel, channelMember) => {
            // Action when clicked
            console.log('Clicked for channel:', channel.id);
        },
        'Settings',        // Dropdown text
        'Plugin Settings'  // Tooltip
    );

    // Register a channel header icon (left side)
    registry.registerChannelHeaderIcon(
        <MyCustomIcon />
    );
    ```
  </Tab>

  <Tab title="Post Components">
    Render custom post types:

    ```typescript
    // Register custom post type
    registry.registerPostTypeComponent(
        'custom_poll',
        PollPostComponent
    );

    // Example component
    const PollPostComponent = ({post}) => {
        const poll = JSON.parse(post.props.poll_data);
        
        return (
            <div className="poll-post">
                <h4>{poll.question}</h4>
                {poll.options.map((option, i) => (
                    <button key={i} onClick={() => vote(option)}>
                        {option}
                    </button>
                ))}
            </div>
        );
    };
    ```

    Create posts with custom type from server:

    ```go
    post := &model.Post{
        Type:      "custom_poll",
        ChannelId: channelID,
        Props: map[string]interface{}{
            "poll_data": pollJSON,
        },
    }
    ```
  </Tab>

  <Tab title="Main Menu">
    Add items to the main menu:

    ```typescript
    registry.registerMainMenuAction(
        'My Plugin',              // Text
        () => {                   // Action
            // Open modal or navigate
            this.openPluginModal();
        },
        <i className="icon fa fa-puzzle-piece"/>  // Mobile icon
    );
    ```
  </Tab>

  <Tab title="Right-Hand Sidebar">
    Display custom content in the RHS:

    ```typescript
    // Register RHS component
    registry.registerRightHandSidebarComponent(
        MyRHSComponent,
        'My Plugin Sidebar'

    // Show the RHS
    registry.registerChannelHeaderButtonAction(
        <i className="icon fa fa-sidebar"/>,

    // RHS Component
    const MyRHSComponent = () => {
        return (
            <div className="plugin-rhs">
    ```

Here are the key registration methods:

<AccordionGroup>
  <Accordion title="registerRootComponent">

  </Accordion>

  <Accordion title="registerPostTypeComponent">
    Render custom post types with specific type prefix.

    ```typescript
    registry.registerPostTypeComponent('custom_type', Component);
    ```
  </Accordion>

  <Accordion title="registerMainMenuAction">
    Add item to main menu.

    ```typescript
    registry.registerMainMenuAction(text, action, mobileIcon);
    ```
  </Accordion>

  <Accordion title="registerChannelHeaderMenuAction">
    Add item to channel menu dropdown.

    ```typescript
    registry.registerChannelHeaderMenuAction(text, action, shouldRender);
    ```
  </Accordion>

  <Accordion title="registerRightHandSidebarComponent">
    Register component for right-hand sidebar.

    ```typescript
    registry.registerRightHandSidebarComponent(Component, title);
    ```
  </Accordion>

  <Accordion title="registerPostDropdownMenuAction">
    Add item to post action menu.

    ```typescript
    registry.registerPostDropdownMenuAction(text, action, icon);
    ```
  </Accordion>

  <Accordion title="registerPostWillRenderEmbedComponent">
    Register component for custom link previews.

    ```typescript
    registry.registerPostWillRenderEmbedComponent(match, Component, toggleable);
    ```
  </Accordion>

  <Accordion title="registerWebSocketEventHandler">
    Handle WebSocket events from server.

    ```typescript
    registry.registerWebSocketEventHandler(
        'custom_com.example.myplugin_event',
        (event) => {
            console.log('Received event:', event);
        }
    );
    ```
  </Accordion>

  <Accordion title="registerSlashCommandWillBePostedHook">
    Intercept slash commands before execution.

    ```typescript
    registry.registerSlashCommandWillBePostedHook((message, args) => {
        if (message.startsWith('/mycommand')) {
            // Handle or modify
            return {message: '/modified'};
        }
        return {message};
    });
    ```
  </Accordion>

  <Accordion title="registerReducer">
    Add custom Redux reducer.

    ```typescript
    registry.registerReducer(myReducer);
    ```
  </Accordion>

  <Accordion title="registerAdminConsoleCustomSection">
    Add custom section to System Console.

    ```typescript
    registry.registerAdminConsoleCustomSection(
        'plugins/com.example.myplugin/settings',
        MySettingsComponent
    );
    ```
  </Accordion>

## Working with Redux

Plugins can access and extend the Redux store:

### Reading State

```typescript
import {getCurrentChannel} from 'mattermost-redux/selectors/entities/channels';
import {getCurrentUser} from 'mattermost-redux/selectors/entities/users';

export default class Plugin {
    public initialize(registry: any, store: Store) {
        this.store = store;
        
        // Access state
        const state = store.getState();
        const currentChannel = getCurrentChannel(state);
        const currentUser = getCurrentUser(state);
    }
}
```

### Custom Reducers

```typescript
// reducers.ts
import {combineReducers} from 'redux';

const myData = (state = {}, action) => {
    switch (action.type) {
        case 'PLUGIN_DATA_RECEIVED':
            return {...state, ...action.data};
        default:
            return state;
    }
};

export default combineReducers({
    myData,
});

// Register in index.tsx
registry.registerReducer(reducer);
```

### Dispatching Actions

```typescript
// actions.ts
export const fetchPluginData = () => async (dispatch) => {
    const response = await fetch('/plugins/com.example.myplugin/api/data');
    const data = await response.json();
    
    dispatch({
        type: 'PLUGIN_DATA_RECEIVED',
        data,
    });
};

// Use in component
store.dispatch(fetchPluginData());
```

## Making API Requests

### Calling Server Plugin Endpoints

```typescript
const callPluginAPI = async (endpoint: string, data: any) => {
    const response = await fetch(
        `/plugins/com.example.myplugin${endpoint}`,
        {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify(data),
        }
    );
    
    if (!response.ok) {
        throw new Error(`API error: ${response.status}`);
    }
    
    return response.json();
};

// Usage
const result = await callPluginAPI('/api/dosomething', {param: 'value'});
```

### Using Mattermost Client

```typescript
import {Client4} from 'mattermost-redux/client';

// Get current user
const user = await Client4.getMe();

// Get posts in channel
const posts = await Client4.getPosts(channelId);

// Create post
const post = await Client4.createPost({
    channel_id: channelId,
    message: 'Hello from plugin',
});
```

## WebSocket Events

### Listening to Events

```typescript
registry.registerWebSocketEventHandler(
    'custom_com.example.myplugin_update',
    (event) => {
        const data = event.data;
        // Update UI based on event
        store.dispatch(updatePluginData(data));
    }
);
```

### Sending Events from Server

From your Go plugin:

```go
p.API.PublishWebSocketEvent("update", map[string]interface{}{
    "message": "Data updated",
}, &model.WebsocketBroadcast{
    ChannelId: channelID,
})
```

Clients receive as: `custom_com.example.myplugin_update`

## Component Examples

### Modal Component

```typescript
import React, {useState} from 'react';
import {Modal} from 'react-bootstrap';

const MyPluginModal = ({show, onHide}) => {
    const [value, setValue] = useState('');
    
    const handleSubmit = async () => {
        await callPluginAPI('/api/submit', {value});
        onHide();
    };
    
    return (
        <Modal show={show} onHide={onHide}>
            <Modal.Header closeButton>
                <Modal.Title>My Plugin</Modal.Title>
            </Modal.Header>
            <Modal.Body>
                <input
                    type="text"
                    value={value}
                    onChange={(e) => setValue(e.target.value)}
                    className="form-control"
                />
            </Modal.Body>
            <Modal.Footer>
                <button onClick={onHide} className="btn btn-link">
                    Cancel
                </button>
                <button onClick={handleSubmit} className="btn btn-primary">
                    Submit
                </button>
            </Modal.Footer>
        </Modal>
    );
};
```

### Interactive Post Component

```typescript
const TaskPost = ({post}) => {
    const [completed, setCompleted] = useState(post.props.completed);
    
    const toggleComplete = async () => {
        await fetch(`/plugins/tasks/api/toggle/${post.id}`, {
            method: 'POST',
        });
        setCompleted(!completed);
    };
    
    return (
        <div className="task-post">
            <input
                type="checkbox"
                checked={completed}
                onChange={toggleComplete}
            />
            <span className={completed ? 'completed' : ''}>
                {post.props.task_text}
            </span>
        </div>
    );
};

registry.registerPostTypeComponent('custom_task', TaskPost);
```
